TC -> O(n) -> one loop
SC -> O(1) -> single varible


class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int n = nums.size();
        int x = 0; 

        for(int num : nums)
        {
            if(num == 0)
                continue;
            else
                nums[x++] = num;
        }

        while(x<n)
            nums[x++] = 0;

    }
};
